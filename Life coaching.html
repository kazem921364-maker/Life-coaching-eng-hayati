function calculateResults(){
const resultsArray = [];
let totalYes = 0;
let resultText = "";
areas.forEach(area => {
let yesCount = 0;
for(let i=1;i<=questionsPerArea;i++){
const val = document.querySelector(`[name='${area}_q${i}']`).value;
if(val==='yes') yesCount++;
}
totalYes += yesCount;
const urgent = yesCount>=3 ? "✅ ﻧﯿﺎﺯ ﻓﻮﺭﯼ" : "❌ ﻧﯿﺎﺯ ﻓﻮﺭﯼ ﻧﺪﺍﺭﺩ";
resultText += `${area}: ${yesCount} ﺑﻠﻪ - ${urgent}\n`;
resultsArray.push({area, yes: yesCount, urgent});
});
resultText += `\nﻣﺠﻤﻮﻉ ﮐﻞ ﺑﻠﻪﻫﺎ: ${totalYes}\n`;
resultText += totalYes>30 ? "📌 ﭘﯿﺸﻨﻬﺎﺩ: ﺷﺮﮐﺖ ﺩﺭ ﮐﻞ ﺩﻭﺭﻩ ﺗﻮﺻﯿﻪ ﻣﯽﺷﻮﺩ" : "📌 ﭘﯿﺸﻨﻬﺎﺩ: ﻧﯿﺎﺯ ﺑﻪ ﺁﻣﻮﺯﺵ ﺩﺭ ﺣﻮﺯﻩﻫﺎﯼ ﻣﺸﺨﺺ ﺷﺪﻩ";
document.getElementById('resultDiv').innerText = resultText;

}
